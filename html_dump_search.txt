            function handleTabKey(e) {
                if (e.key === 'Tab' || e.keyCode === 9) {
                    const $items = $popup.find(focusSelector).filter(':visible');
                    const $first = $items.first();
                    const $last = $items.last();

                    const $active = $(':focus');

                    if (e.shiftKey) { // Shift + Tab
                        // 현재 포커스가 첫 번째 요소인 경우
                        if ($active.is($first)) {
                            e.preventDefault();
                            $last.focus(); // 마지막 요소로 이동
                        }
                    } else { // Tab
                        // 현재 포커스가 마지막 요소인 경우
                        if ($active.is($last)) {
                            e.preventDefault();
                            $first.focus(); // 첫 번째 요소로 이동
                        }
                    }
                }

                    <p class="grayText txt01 mt40">제2조(정의)</p>
                    <p class="txt01 mt10">본 약관에서 사용하는 용어의 정의는 다음과 같습니다.</p>
                    <p class="txt01 mt10">1. "이용자"란 국회 홈페이지에 접속하여 본 약관에 따라 국회가 제공하는 서비스를 이용하는 회원 및 비회원을 말합니다.</p>
                    <p class="txt01 mt10">2. "회원"이란 국회 홈페이지의 가입 신청 절차에 따라 회원 등록을 한 자로서 국회 홈페이지의 정보 및 서비스를 계속적으로 이용할 수 있는 자를 말합니다.</p>
                    <p class="txt01 mt10">3. "본인확인"이란 국회 홈페이지가 제공하는 서비스 이용을 위하여 공공/일반 I-PIN 인증 또는 휴대폰 번호 인증 등으로 본인임을 확인하는 것을 말합니다.</p>
                    <p class="grayText txt01 mt40">제3조(약관의 효력과 개정)</p>
                    <p class="txt01 mt10">① 본 약관은 국회 홈페이지 서비스 화면에 공지함으로써 이용자에게 공시하고, 본 약관에 동의한 이용자가 회원에 가입함으로써 효력이 발생하며, 국회 홈페이지의 서비스 제공 행위 및 이용자의 서비스 사용 행위에 본 약관이 우선적으로 적용됩니다.</p>
                    <p class="txt01 mt10">② 국회는 필요한 경우 이 약관의 내용을 변경할 수 있으며, 변경된 약관은 국회 홈페이지 서비스 화면에 공지함으로써 이용자가 직접 확인할 수 있도록 합니다.</p>
                    <p class="txt01 mt10">③ 회원은 개정된 약관에 동의하지 않을 경우 회원 탈퇴할 수 있으며, 명시적으로 거부의 의사표시를 하지 아니하고 계속 사용하는 경우에는 약관 변경에 동의한 것으로 간주됩니다.</p>
                    <p class="txt01 mt10">④ 본 약관에 명시되지 않은 사항에 대해서는 「전기통신사업법」, 「정보통신망 이용촉진 및 정보보호 등에 관한 법률」, 「개인정보 보호법」 등 관련 법령의 규정에 따릅니다.</p>
                    <h3 class="title01 mt35 tac">제2장 서비스 제공 및 이용</h3>
                    <p class="grayText txt01 mt40">제4조(회원 가입)</p>
                    <p class="txt01 mt10">① 국회 홈페이지 회원에 가입하고자 하는 이용자는 국회 홈페이지의 회원 가입 신청 절차에 따라 본인확인을 거쳐 본 이용약관 및 개인정보처리방침 등에 "동의" 여부를 표시하고, 개인정보를 기재함으로써 회원에 가입합니다.</p>
                    <p class="grayText txt01 mt40">제5조(회원 탈퇴 및 자격 상실)</p>
                    <p class="txt01 mt10">① 회원은 언제든지 회원 탈퇴를 할 수 있으며 국 회는 지체 없이 회원 탈퇴를 처리하고, 해당 개인정보를 파기합니다.</p>
                    <p class="txt01 mt10">② 국회 홈페이지가 제12조제2항 및 제13조제1항에 따라 이용자의 회원 자격을 상실시키기로 결정한 경우에는 회원 등록을 말소합니다. 이 경우 회원 등록 말소 전에 회원에게 이를 통지하고, 소명할 기회를 부여합니다.</p>
                    <p class="grayText txt01 mt40">제6조(서비스의 제공 및 변경)</p>
                    <p class="txt01 mt10">① 국회 홈페이지는 이용자에게 다음과 같은 서비스를 제공합니다.</p>
                    <p class="txt01 mt10">1. 국회의 의정활동 관련 콘텐츠, 맞춤서비스, 홍보행사 정보 및 참여기회 제공 등의 대국민 정보 서비스</p>
                    <p class="txt01 mt10">2. 온라인 상의 민원 처리 서비스</p>
                    <p class="txt01 mt10">② 국회 홈페이지는 서비스의 내용이 변경되는 경우, 그 내용 및 제공 일자를 제8조제2항에서 정한 방법으로 이용자에게 통지하고, 제1항에 정한 서비스를 변경하여 제공할 수 있습니다.</p>
                    <p class="grayText txt01 mt40">제7조(서비스의 중단)</p>
                    <p class="txt01 mt10">① 국회 홈페이지는 정보시스템, 서버, 정보기기, 네트워크의 점검·교체 및 장애발생 등의 사유가 발생한 경우에는 서비스의 제공을 일시적으로 중단할 수 있습니다. 또한, 운영상 상당한 이유가 있는 경우 현재 제공되는 서비스를 새로운 서비스로 교체 또는 중단할 수 있습니다.</p>
                    <p class="txt01 mt10">② 제1항에 의한 서비스 중단의 경우에는 국회는 제8조제2항에서 정한 방법으로 이용자에게 통지합니다. 다만, 국회가 통제할 수 없는 사유로 인한 서비스의 중단(시스템 관리자의 고의·과실이 없는 정보시스템, 서버, 정보기기, 네트워크의 장애발생 등)으로 인하여 사전 통지가 불가능한 경우에는 그러하지 아니합니다.</p>
                    <p class="grayText txt01 mt40">제8조(이용자에 대한 통지)</p>
                    <p class="txt01 mt10">① 국회가 특정 이용자에 대한 통지를 하는 경우 이용자가 회원 가입 시에 기재한 메일주소로 할 수 있습니다.</p>
                    <p class="txt01 mt10">② 국회가 불특정다수 이용자에 대한 통지를 하는 경우 국회 홈페이지 게시판 및 서비스 화면에 게시함으로써 개별 통지에 갈음할 수 있습니다.</p>
                    <p class="grayText txt01 mt40">제9조(개인정보보호)</p>
                    <p class="txt01 mt10">① 국회는 「개인정보 보호법」등 관련 법령, 국회 개인정보 보호에 관한 규정 및 국회의 개인정보처리방침이 정하는 바에 따라 이용자의 개인정보를 보호하기 위하여 노력합니다. 다만, 국회가 운영하는 홈페이지에 포함된 링크 또는 배너를 클릭하여 다른 사이트로 옮겨갈 경우에는 해당 사이트의 개인정보보호방침에 따릅니다.</p>
                    <p class="txt01 mt10">② 국회 홈페이지는 회원 정보를 다음과 같이 처리합니다.</p>
                    <p class="txt01 mt10">1. 개인정보의 수집 및 이용 : 국회 홈페이지는 회원가입 시, 대국민 서비스 제공·민원처리·소관업무 수행 등의 목적에 필요한 범위 내에서 이용자 본인의 동의(14세 미만인 경우 법정 대리인의 동의)를 받아 최소한의 개인정보를 수집합니다. 또한, 민감정보·고유식별정보의 수집이 필요한 경우에는 각각을 구분하여 별도의 동의를 받아 처리합니다.</p>
                    <p class="txt01 mt10">2. 개인정보의 목적 외 이용 및 제3자 제공 : 국회 홈페이지는 수집된 회원의 개인정보를 제2항제1호의 목적 범위 내에서 이용하며, 이용자 본인의 동의 없이 본래의 목적을 초과하여 처리하거나 제3자에게 제공하지 않습니다.</p>
                    <p class="txt01 mt10">③ 회원은 개인정보 관리화면을 통하여 언제든지 본인의 개인정보를 열람·수정·삭제할 수 있습니다.</p>
                    <p class="txt01 mt10">④ 회원은 회원 가입 신청 시 기재한 사항에 변경이 발생한 경우, 변경사항을 수정하여 기재하여야 합니다. 회원이 변경사항을 수정 기재하지 않아 발생한 불이익에 대하여 국회는 책임지지 않습니다.</p>
                    <p class="grayText txt01 mt40">제10조(국회의 의무)</p>
                    <p class="txt01 mt10">① 국회는 관련 법령과 본 약관이 정하는 바에 따라 지속적이고, 안정적으로 서비스를 제공하기 위해서 노력합니다.</p>
                    <p class="txt01 mt10">② 국회는 이용자가 안전하게 인터넷 서비스를 이용할 수 있도록 개인정보의 안전성 확보에 필요한 기술적·관리적 및 물리적 보안 조치를 취합니다.</p>
                    <p class="grayText txt01 mt40">제11조(이용자의 ID 및 비밀번호에 대한 의무)</p>
                    <p class="txt01 mt10">① 이용자의 ID와 비밀번호에 관한 관리책임은 이용자에게 있습니다. 다만, 국회가 개인정보보호법 등 관계 법령에 의하여 책임을 지는 경우는 제외합니다.</p>
                    <p class="txt01 mt10">② 이용자는 자신의 ID 및 비밀번호를 제3자에게 이용하게 해서는 안 되며, 작업 종료 시에는 반드시 로그아웃하고, 웹브라우저의 창을 닫아야 합니다.</p>
                    <p class="txt01 mt10">③ 이용자는 자신의 ID 및 비밀번호가 도용되거나 제3자가 사용하고 있음을 인지한 경우에는 바로 국회에 통보하고 국회의 안내가 있는 경우에는 그에 따라야 합니다.</p>
                    <p class="grayText txt01 mt40">제12조(이용자의 의무)</p>
                    <p class="txt01 mt10">①이용자는 다음 각 호의 행위를 하여서는 안 됩니다.</p>
                    <p class="txt01 mt10">1. 신청 또는 변경 시 허위 내용을 등록하는 행위</p>
                    <p class="txt01 mt10">2. 타인의 명의를 도용하여 부정사용하는 행위</p>
                    <p class="txt01 mt10">3. 국회의 직원이나 국회 서비스의 관리자를 가장하거나 사칭하는 행위</p>
                    <p class="txt01 mt10">4. 국회 홈페이지에 게시된 정보를 변경하는 행위</p>
                    <p class="txt01 mt10">5. 다른 이용자에 대한 개인정보를 수집·저장·공개하는 행위</p>
                    <p class="txt01 mt10">6. 외설 또는 폭력적·위협적인 메시지·화상·음성·기타 공서양속에 반하는 정보를 메일로 송신하거나 공개 또는 게시하는 행위</p>
                    <p class="txt01 mt10">7. 사생활 침해 또는 명예훼손 등 타인의 권리를 침해하는 정보를 유통시키는 행위</p>
                    <p class="txt01 mt10">8. 국회 또는 타인의 지적재산권을 침해하거나 업무를 방해하는 행위</p>
                    <p class="txt01 mt10">9. 컴퓨터 소프트웨어·하드웨어·네트워크 등의 정상적인 가동을 방해, 파괴할 목적으로 고안된 소프트웨어 바이러스·기타 다른 컴퓨터 코드·파일·프로그램을 포함하고 있는 자료를 게시하거나 전자우편으로 발송하는 행위</p>
                    <p class="txt01 mt10">10. 불특정 다수를 대상으로 하여 광고 또는 선전을 게시하거나 스팸메일을 전송하는 등의 행위</p>
                    <p class="txt01 mt10">11. 국회 홈페이지의 이용 약관을 위반하는 행위</p>
                    <p class="txt01 mt10">12. 기타 관련 법령에 의하여 그 전송 또는 게시가 금지되는 정보를 전송 또는 게시하는 행위</p>
                    <p class="txt01 mt10">② 제1항에 해당하는 행위를 한 이용자가 있을 경우 국회는 이용자의 회원자격을 적정한 방법으로 제한, 정지 또는 상실시킬 수 있습니다.</p>
                    <p class="txt01 mt10">③ 이용자는 그 귀책사유로 인하여 국회나 다른 이용자가 입은 손해를 배상할 책임이 있습니다.</p>
                    <p class="grayText txt01 mt40">제13조(공개 게시물의 삭제 등)</p>
                    <p class="txt01 mt10">① 이용자가 게재한 공개 게시물의 내용이 다음 각 호에 해당하는 경우 국회 홈페이지는 이용자에게 사전 통지 없이 해당 공개 게시물을 삭제할 수 있고, 해당 이용자의 회원 자격을 제한, 정지 또는 상실시킬 수 있습니다.</p>
                    <p class="txt01 mt10">1. 음란한 부호·문언·음향·화상 또는 영상을 배포·판매·임대하거나 공공연하게 전 시하는 내용</p>
                    <p class="txt01 mt10">2. 사람을 비방할 목적으로 공공연하게 사실이나 거짓의 사실을 드러내어 타인의 명예를 훼손하는 내용</p>
                    <p class="txt01 mt10">3. 공포심이나 불안감을 유발하는 부호·문언·음향·화상 또는 영상을 반복적으로 상대방에게 도달하도록 하는 내용</p>
                    <p class="txt01 mt10">4. 정당한 사유 없이 정보통신시스템, 데이터 또는 프로그램 등을 훼손·멸실·변경·위조하거나 그 운용을 방해하는 내용</p>
                    <p class="txt01 mt10">5. 「청소년 보호법」에 따른 청소년유해매체물을 게재 또는 광고하는 내용</p>
                    <p class="txt01 mt10">6. 법령에 따라 금지되는 사행행위에 해당하는 내용</p>
                    <p class="txt01 mt10">7. 법령에 따라 분류되는 비밀 등 국가기밀을 누설하는 내용</p>
                    <p class="txt01 mt10">8.「국가보안법」에서 금지하는 행위를 수행하는 내용</p>
                    <p class="txt01 mt10">9. 그 밖에 범죄를 목적으로 하거나 교사 또는 방조하는 내용</p>
                    <p class="txt01 mt10">10. 종교적 분쟁 등을 야기하는 내용으로서, 이러한 분쟁으로 인하여 국회의 업무가 방해되거나 방해되리라고 판단되는 경우</p>
                    <p class="txt01 mt10">11. 기타 관련 법령에 위배된다고 판단되는 내용</p>
                    <p class="txt01 mt10">② 공개게시물의 내용으로 인하여 사생활 침해나 명예훼손 등 타인의 권리가 침해된 경우 그 침해를 받은 자는 국회에게 침해사실을 소명하여 그 정보의 삭제를 요청할 수 있습니다.</p>
                    <p class="txt01 mt10">③ 국회는 제2항에 따른 요청을 받으면 지체 없이 삭제·임시조치 등의 필요한 조치를 하고, 즉시 신청인 및 정보 게시자에게 알리는 한편, 필요한 조치를 취한 사실을 해당 게시판에 공시하는 등의 방법으로 이용자가 알 수 있도록 합니다.</p>
                    <p class="txt01 mt10">④ 국회는 제2항에 따른 정보의 삭제요청에도 불구하고, 권리의 침해 여부를 판단하기 어렵거나 이해당사자 간에 다툼이 예상되는 경우에는 해당 정보에 대한 접근을 임시적으로 차단하는 조치(이하 "임시조치"라 한다)를 할 수 있습니다. 이 경우 임시조치의 기간은 30일 이내로 합니다.</p>
                    <p class="grayText txt01 mt40">제14조(저작권의 귀속 및 공공저작물의 자유이용)</p>
                    <p class="txt01 mt10">① 국회가 창작·작성한 저작물에 대한 저작권 및 기타 지적재산권은 국회사무처에 귀속합니다.</p>
                    <p class="txt01 mt10">② 저작권법 제24조의2(공공저작물의 자유이용)에 따라 국회에서 제공하는 자료 중에서 국회사무처가 저작재산권의 전부를 보유한 저작물의 경우에는 이용허락 없이 자유이용이 가능합니다. 자유이용이 가능한 자료는 "공공저작물 자유이용허락 표시 기준"에 따라 "공공누리, KOGL 제1유형"을 부착하여 개방하고 있습니다.</p>
                    <p class="txt01 mt10">③ 이용자는 공공누리 표시가 부착된 공공저작물을 상업적 활용 여부에 관계없이 무료로 자유롭게 이용하고 2차적 저작물 작성 등 변형하여 이용할 수 있습니다. 다만, 이용자는 저작물의 출처를 반드시 표시해야할 의무가 있으며, 개인정보보호법 등 다른 법령에 따라 이용이 제한되는 정보의 경우에는 공공저작물의 이용이 제한됩니다.</p>
                    <h3 class="title01 mt35 tac">제3장 손해배상 등</h3>
                    <p class="grayText txt01 mt40">제15조(손해배상)</p>
                    <p class="txt01 mt10">국회는 무료로 제공되는 서비스와 관련하여 회원에게 어떠한 손해가 발생하더라도 동 손해가 국회의 중대한 과실에 의한 경우를 제외하고는 이에 대하여 책임을 부담하지 않습니다.</p>
                    <p class="grayText txt01 mt40">제16조(면책조항)</p>
                    <p class="txt01 mt10">① 국회는 천재지변 또는 이에 준하는 불가항력으로 인하여 서비스를 제공할 수 없는 경우에는 서비스 제공에 관한 책임이 면제됩니다.</p>
                    <p class="txt01 mt10">② 국회는 회원의 귀책사유로 인한 서비스 이용의 장애에 대하여 책임을 지지 않습니다.</p>
                    <p class="txt01 mt10">③ 국회는 회원이 서비스를 이용하여 기대하는 이익이나 서비스를 통하여 얻은 자료로 인한 손해에 관하여 책임을 지지 않습니다.</p>
                    <p class="txt01 mt10">④ 국회는 회원이 서비스에 게재한 정보, 자료, 사실의 신뢰도, 정확성 등의 내용에 관하여는 책임을 지지 않습니다.</p>
                    <p class="grayText txt01 mt40">제17조(재판관할)</p>
                    <p class="txt01 mt10">① 서비스 이용과 관련하여 국회와 회원 사이에 분쟁이 발생한 경우, 국회와 회원은 발생한 분쟁을 원만하게 해결하기 위하여 필요한 모든 노력을 하여야 합니다.</p>
                    <p class="txt01 mt10">② 제1항의 규정에도 불구하고 서비스 이용으로 발생한 분쟁에 대하여 소송이 제기될 경우 국회 소재지를 관할하는 법원을 관할법원으로 합니다.</p>

        function footer_loadMyContentList() {
            const renderList = () => {
                const myContentList = getList();
                const $sect = $("#footer_myContentList");
                $sect.empty();

                if (myContentList != null && myContentList.length > 0) {
                    _.forEach(myContentList, (item) => {
                        addItem($sect, item);
                    });

                    initSwiper();
                }
                else {
                    $("#footer_myContentList").append("<div class='noMymemoData'>내가 본 콘텐츠가 없습니다.</div>");
                    $(".mymemo_box .number_btn").hide();
                }
            }

            const getList = () => {
                return gfn_getAllFromLocalStorage("myContentList")
            }

            const addItem = ($sect, data) => {
                let urlLink = removeEmptyParams(data.urlLink);
                // console.log(urlLink)
                const item =
                    "<div class=\"swiper-slide\">" +
                    "   <a href='" + com.help.url(urlLink) + "' title='" + data.title + "'>" +
                    "      <em>" + data.divName + "</em>" +
                    "      <span>" + data.title + "</span>" +
                    "   </a>"+
                    "</div>";
                $sect.append(item);
            }

            const removeEmptyParams = function(url) {
                // URL을 쿼리스트링 기준으로 분리
                const [baseUrl, queryString] = url.split('?');
                if (!queryString) return baseUrl; // 쿼리스트링이 없는 경우 그대로 반환

                // 쿼리스트링을 '&'로 나누고 값이 있는 파라미터만 필터링
                const filteredParams = queryString
                    .split('&')
                    .filter(param => {
                        const [key, value] = param.split('=');
                        return key && value; // key와 value가 모두 존재해야 포함
                    });

                // 필터링된 파라미터를 쿼리스트링으로 재조합
                const filteredQueryString = filteredParams.join('&');

                // 파라미터가 남아있으면 baseUrl에 추가, 없으면 baseUrl만 반환
                return filteredQueryString ? `${baseUrl}?${filteredQueryString}` : baseUrl;
            }

            const initSwiper = () => {
                var swiper = new Swiper('#footer_myContentList_swiper', {
                    slidesPerView: 1,
                    slidesPerGroup: 1,	//한번에 슬라이딩될 개수
                    grid: {
                        rows: 3,
                    },
                    init: false,
                    freeMode : false, // 슬라이드 넘길 때 위치 고정 여부
                    loop: false,
                    spaceBetween: 0,
                    navigation: {
                        prevEl: '.nb_prev',
                        nextEl: '.nb_next',
                    },
                    pagination: {
                        el: '#footer_myContentList_pager',
                        type: "fraction",
                        clickable: true,
                    },
                    scrollbar: {
                        hide:true
                    },
                    a11y:{
                        enabled:true,
                    },
                    autoplay: false,
                    observer: true,
                    observeParents: true,
                });
                swiper.init();//초기화
            }

            renderList();   // 목록생성
        }

        function footer_moveToBillDetailPageByBillNo(billNo) {
            // 한눈에 보는 통계 TOP5 바로가기 추가
            if (com.util.str.isNull(billNo)) {
                // 기존 의안 바로가기
                billNo = $("#footer_billNo__txt").val();
                if (com.util.str.isNull(billNo)) {
                    alert("의안번호를 입력하세요.");
                    return false;
                }
                // 17대이하는 6자리, 18대이상은 7자리 의안번호
                if (!/^((1[89]|[2-9][0-9])\d{5}|(0[1-9]|1[0-7])\d{4})$/.test(billNo)) {
                    alert("정확한 의안번호를 입력하세요.");
                    return false;
                }
            }
            com.help.ajax({
                url: "/bi/common/findBillDetail.do",
                params: {billNo: billNo},
                done: function(res) {
                    if ( res.data && res.data.billId ) {
                        const billUrl = com.help.url(`/bi/billDetailPage.do?billId=${res.data.billId}`);
                        window.open(billUrl, "_blank");
                    } else {
                        alert("해당 의안번호가 존재하지 않습니다.");
                    }

                }
            });
        }

        /**
         * 푸터 관련 사이트 통합 로드 (AJAX)
         * - 소관기관 (JRSD_HMPG_CD)
         * - 위원회 (CMMIT_HMPG_CD)
         * - 홈페이지 (HMPG_CD)
         * - 국회정보시스템 (NA_INFO_SYS_CD)
         */
        function footer_loadRelatedSites() {
            $.ajax({
                url: "https://assembly.go.kr/portal/api/footerInfo.do",
                type: "GET",
                dataType: "json",
                success: function(data) {
                    // 데이터가 없으면 중단
                    if (!data || data.length === 0) return;

                    // 각 select 박스 객체
                    var $jrsdSelect = $("#footer_jrsd_hmpg_cd");      // 소관기관
                    var $cmmitSelect = $("#footer_cmmit_hmpg_cd");    // 위원회
                    var $hmpgSelect = $("#footer_hmpg_cd");           // 홈페이지
                    var $infoSelect = $("#footer_na_info_sys_cd");    // 국회정보시스템

                    $jrsdSelect.find("option:not(:first)").remove();
                    $cmmitSelect.find("option:not(:first)").remove();
                    $hmpgSelect.find("option:not(:first)").remove();
                    $infoSelect.find("option:not(:first)").remove();

                    $.each(data, function(index, item) {
                        var $targetSelect = null;

                        // grpComCd 코드로 분기 처리
                        switch (item.grpComCd) {
                            case "JRSD_HMPG_CD":    // 소관기관
                                $targetSelect = $jrsdSelect;
                                break;
                            case "CMMIT_HMPG_CD":   // 위원회
                                $targetSelect = $cmmitSelect;
                                break;
                            case "HMPG_CD":         // 홈페이지
                                $targetSelect = $hmpgSelect;
                                break;
                            case "NA_INFO_SYS_CD":  // 국회정보시스템
                                $targetSelect = $infoSelect;
                                break;
                            default:
                                // 정의되지 않은 코드는 무시
                                break;
                        }

                        // 옵션 추가
                        if ($targetSelect) {
                            var $option = $("<option></option>")
                                .attr("value", item.url)
                                .text(item.comNm);
                            $targetSelect.append($option);
                        }
                    });
                },
                error: function(xhr, status, error) {
                    console.error("관련 사이트 정보 로드 실패:", error);
                }
            });
        }
    </script>
