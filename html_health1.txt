/**
 * 내부IP 목록 조회
 */
function glnbInnerIpCheck() {
    
    var innerIpYn = false;
    if ( INNER_IP_BAND_LIST == null || INNER_IP_BAND_LIST == undefined || INNER_IP_BAND_LIST.length == 0 ) {
        var params = {
              menuNo   : $("#menuNo").val()
            // Spring Security CSRF Param 처리
            , _csrf    : $("meta[name='_csrf']").attr("content")
        };
        $.ajax({
            type    : 'post',
            url     : "/cmmit/main/innerIpBandList.json",
            data    : params,
            async   : false,
            success : function(data) {
                var resultCode = data.resultCode;
                var resultMsg  = data.msg;
                if(resultCode == "success") {
                    var resultList = data.ipList;
                    INNER_IP_BAND_LIST = [];
                    if ( resultList != null && resultList.length > 0 ) {
                        for ( var i = 0 ; i < resultList.length ; i++ ) {
                            if ( resultList[ i ].com1Val != "" && resultList[ i ].com1Val != null ) {
                                INNER_IP_BAND_LIST.push(decodeNaValue(resultList[ i ].com1Val));
                            }
                        }

                        innerIpYn = glnbInnerIpSubCheck();
                    } else {
                        innerIpYn = false;
                    }
                }
            }
        });
    } else {
        innerIpYn = glnbInnerIpSubCheck();
    }
    
    return innerIpYn;
}

/**
 * 접속 IP 가 내부IP 인지 확인.. 
 */
function glnbInnerIpSubCheck() {
    
    var innerIpYn = false;
    var innerIpList = INNER_IP_BAND_LIST;
    
//     console.log("------------------------------");
//     console.log("INNER_IP_BAND_LIST::" + JSON.stringify(INNER_IP_BAND_LIST));
//     console.log("------------------------------");
    
    if ( innerIpList != null && innerIpList.length > 0 ) {
        for ( var i = 0 ; i < innerIpList.length ; i++ ) {
            var idx = innerIpList[i].indexOf(".*");
            if ( idx >= 0 ) {
                // 내부 IP 대역 확인
                var innIpBand = innerIpList[i].substring(0, idx);
                if ( HOME_PAGE_MY_ADDR.indexOf(innIpBand) == 0 ) {
                    innerIpYn = true;
                    break;
                }
            } else {
                // 내부 IP 확인
                if ( HOME_PAGE_MY_ADDR == innerIpList[i] ) {
                    innerIpYn = true;
                    break;
                }
            }
        }
    } else {
        innerIpYn = false;
    }
    
    // console.log("inner ip YN::"+innerIpYn);
    return innerIpYn;
}

/**
 * 입력된 암호된 값을 복호화 처리
 *
 * @param encodeStr
 * @returns
 */
function decodeNaValue(encodeStr)  {

    var decodeBase64Var = '';
    try {
        decodeBase64Var = atob(encodeStr);
    } catch(e) {
        decodeBase64Var = encodeStr;
    }

    var len = decodeBase64Var.length;
    var decodeStr = '';
    for ( var i = 0 ; i < len ; i ++ ) {
        var numChr = decodeBase64Var.charCodeAt(i);
        var decodeNumChr = numChr - 3;
        decodeStr = decodeStr + String.fromCharCode(decodeNumChr);
    }

    return decodeStr;
}

                    <p class="txt01 mt10">7. 사생활 침해 또는 명예훼손 등 타인의 권리를 침해하는 정보를 유통시키는 행위</p>
                    <p class="txt01 mt10">8. 국회 또는 타인의 지적재산권을 침해하거나 업무를 방해하는 행위</p>
                    <p class="txt01 mt10">9. 컴퓨터 소프트웨어·하드웨어·네트워크 등의 정상적인 가동을 방해, 파괴할 목적으로 고안된 소프트웨어 바이러스·기타 다른 컴퓨터 코드·파일·프로그램을 포함하고 있는 자료를 게시하거나 전자우편으로 발송하는 행위</p>
                    <p class="txt01 mt10">10. 불특정 다수를 대상으로 하여 광고 또는 선전을 게시하거나 스팸메일을 전송하는 등의 행위</p>
                    <p class="txt01 mt10">11. 국회 홈페이지의 이용 약관을 위반하는 행위</p>
                    <p class="txt01 mt10">12. 기타 관련 법령에 의하여 그 전송 또는 게시가 금지되는 정보를 전송 또는 게시하는 행위</p>
                    <p class="txt01 mt10">② 제1항에 해당하는 행위를 한 이용자가 있을 경우 국회는 이용자의 회원자격을 적정한 방법으로 제한, 정지 또는 상실시킬 수 있습니다.</p>
                    <p class="txt01 mt10">③ 이용자는 그 귀책사유로 인하여 국회나 다른 이용자가 입은 손해를 배상할 책임이 있습니다.</p>
                    <p class="grayText txt01 mt40">제13조(공개 게시물의 삭제 등)</p>
                    <p class="txt01 mt10">① 이용자가 게재한 공개 게시물의 내용이 다음 각 호에 해당하는 경우 국회 홈페이지는 이용자에게 사전 통지 없이 해당 공개 게시물을 삭제할 수 있고, 해당 이용자의 회원 자격을 제한, 정지 또는 상실시킬 수 있습니다.</p>
                    <p class="txt01 mt10">1. 음란한 부호·문언·음향·화상 또는 영상을 배포·판매·임대하거나 공공연하게 전 시하는 내용</p>
                    <p class="txt01 mt10">2. 사람을 비방할 목적으로 공공연하게 사실이나 거짓의 사실을 드러내어 타인의 명예를 훼손하는 내용</p>
                    <p class="txt01 mt10">3. 공포심이나 불안감을 유발하는 부호·문언·음향·화상 또는 영상을 반복적으로 상대방에게 도달하도록 하는 내용</p>
                    <p class="txt01 mt10">4. 정당한 사유 없이 정보통신시스템, 데이터 또는 프로그램 등을 훼손·멸실·변경·위조하거나 그 운용을 방해하는 내용</p>
                    <p class="txt01 mt10">5. 「청소년 보호법」에 따른 청소년유해매체물을 게재 또는 광고하는 내용</p>
                    <p class="txt01 mt10">6. 법령에 따라 금지되는 사행행위에 해당하는 내용</p>
                    <p class="txt01 mt10">7. 법령에 따라 분류되는 비밀 등 국가기밀을 누설하는 내용</p>
                    <p class="txt01 mt10">8.「국가보안법」에서 금지하는 행위를 수행하는 내용</p>
                    <p class="txt01 mt10">9. 그 밖에 범죄를 목적으로 하거나 교사 또는 방조하는 내용</p>
                    <p class="txt01 mt10">10. 종교적 분쟁 등을 야기하는 내용으로서, 이러한 분쟁으로 인하여 국회의 업무가 방해되거나 방해되리라고 판단되는 경우</p>
                    <p class="txt01 mt10">11. 기타 관련 법령에 위배된다고 판단되는 내용</p>
                    <p class="txt01 mt10">② 공개게시물의 내용으로 인하여 사생활 침해나 명예훼손 등 타인의 권리가 침해된 경우 그 침해를 받은 자는 국회에게 침해사실을 소명하여 그 정보의 삭제를 요청할 수 있습니다.</p>
                    <p class="txt01 mt10">③ 국회는 제2항에 따른 요청을 받으면 지체 없이 삭제·임시조치 등의 필요한 조치를 하고, 즉시 신청인 및 정보 게시자에게 알리는 한편, 필요한 조치를 취한 사실을 해당 게시판에 공시하는 등의 방법으로 이용자가 알 수 있도록 합니다.</p>
                    <p class="txt01 mt10">④ 국회는 제2항에 따른 정보의 삭제요청에도 불구하고, 권리의 침해 여부를 판단하기 어렵거나 이해당사자 간에 다툼이 예상되는 경우에는 해당 정보에 대한 접근을 임시적으로 차단하는 조치(이하 "임시조치"라 한다)를 할 수 있습니다. 이 경우 임시조치의 기간은 30일 이내로 합니다.</p>
                    <p class="grayText txt01 mt40">제14조(저작권의 귀속 및 공공저작물의 자유이용)</p>
                    <p class="txt01 mt10">① 국회가 창작·작성한 저작물에 대한 저작권 및 기타 지적재산권은 국회사무처에 귀속합니다.</p>
                    <p class="txt01 mt10">② 저작권법 제24조의2(공공저작물의 자유이용)에 따라 국회에서 제공하는 자료 중에서 국회사무처가 저작재산권의 전부를 보유한 저작물의 경우에는 이용허락 없이 자유이용이 가능합니다. 자유이용이 가능한 자료는 "공공저작물 자유이용허락 표시 기준"에 따라 "공공누리, KOGL 제1유형"을 부착하여 개방하고 있습니다.</p>
                    <p class="txt01 mt10">③ 이용자는 공공누리 표시가 부착된 공공저작물을 상업적 활용 여부에 관계없이 무료로 자유롭게 이용하고 2차적 저작물 작성 등 변형하여 이용할 수 있습니다. 다만, 이용자는 저작물의 출처를 반드시 표시해야할 의무가 있으며, 개인정보보호법 등 다른 법령에 따라 이용이 제한되는 정보의 경우에는 공공저작물의 이용이 제한됩니다.</p>
                    <h3 class="title01 mt35 tac">제3장 손해배상 등</h3>
                    <p class="grayText txt01 mt40">제15조(손해배상)</p>
                    <p class="txt01 mt10">국회는 무료로 제공되는 서비스와 관련하여 회원에게 어떠한 손해가 발생하더라도 동 손해가 국회의 중대한 과실에 의한 경우를 제외하고는 이에 대하여 책임을 부담하지 않습니다.</p>
                    <p class="grayText txt01 mt40">제16조(면책조항)</p>
                    <p class="txt01 mt10">① 국회는 천재지변 또는 이에 준하는 불가항력으로 인하여 서비스를 제공할 수 없는 경우에는 서비스 제공에 관한 책임이 면제됩니다.</p>
                    <p class="txt01 mt10">② 국회는 회원의 귀책사유로 인한 서비스 이용의 장애에 대하여 책임을 지지 않습니다.</p>
                    <p class="txt01 mt10">③ 국회는 회원이 서비스를 이용하여 기대하는 이익이나 서비스를 통하여 얻은 자료로 인한 손해에 관하여 책임을 지지 않습니다.</p>
                    <p class="txt01 mt10">④ 국회는 회원이 서비스에 게재한 정보, 자료, 사실의 신뢰도, 정확성 등의 내용에 관하여는 책임을 지지 않습니다.</p>
                    <p class="grayText txt01 mt40">제17조(재판관할)</p>
                    <p class="txt01 mt10">① 서비스 이용과 관련하여 국회와 회원 사이에 분쟁이 발생한 경우, 국회와 회원은 발생한 분쟁을 원만하게 해결하기 위하여 필요한 모든 노력을 하여야 합니다.</p>
                    <p class="txt01 mt10">② 제1항의 규정에도 불구하고 서비스 이용으로 발생한 분쟁에 대하여 소송이 제기될 경우 국회 소재지를 관할하는 법원을 관할법원으로 합니다.</p>
                    <h3 class="title01 mt35 tac">&lt;부 칙&gt;</h3>
                    <p>본 약관은 2014. 10. 22. 부터 적용합니다.</p>
                    <p>* 본 약관에 대한 저작권은 대한민국 국회에 귀속하며 무단 복제·배포·전송·기타 저작권 침해행위를 엄금합니다.</p>
